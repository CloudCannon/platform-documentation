## Options

&nbsp;

<comp.MultiCodeBlock language="YAML" translate_into={["json"]} source="cloudcannon.config.yml">
``````````

collections_config:
  blog:
    path: content/blog
    glob:
      - '!_index.md'
    name: Blog
    icon: post_add
    disable_url: false
    description: ''
    _editables:
    _enabled_editors:
    _inputs:
    _select_data:
    _structures:
    disable_add: false
    disable_add_folder: false
    disable_file_actions: false
    schemas:
      post:
        path: schemas/post.md
        reorder_inputs: true
        hide_extra_inputs: false
        remove_empty_inputs: false
        remove_extra_inputs: true
        name: Blog Post
        icon: post_add
      customer_story:
        path: schemas/customer-story.md
        reorder_inputs: true
        hide_extra_inputs: false
        remove_empty_inputs: false
        remove_extra_inputs: true
        icon: auto_stories

``````````
